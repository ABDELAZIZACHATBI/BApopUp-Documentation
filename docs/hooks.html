Actions & Filters

Available Actions

bapopup_before_show

Fires before popup is displayed

add_action('bapopup_before_show', function($popup_id) {
                // Your code here
            });

bapopup_after_show

Fires after popup is displayed

add_action('bapopup_after_show', function($popup_id) {
                // Your code here
            });

Available Filters

bapopup_display_conditions

Modify popup display conditions

add_filter('bapopup_display_conditions', function($conditions, $popup_id) {
                // Modify conditions
                return $conditions;
            }, 10, 2);
