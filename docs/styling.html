Custom Styling

../css Examples

Popup Container

.bapopup-container {
                max-width: 500px;
                border-radius: 15px;
                padding: 30px;
                box-shadow: 0 10px 20px rgba(0,0,0,0.1);
            }

Close Button

.bapopup-close {
                position: absolute;
                top: 10px;
                right: 10px;
                background: #f0f0f0;
                border-radius: 50%;
                width: 30px;
                height: 30px;
                line-height: 30px;
                text-align: center;
                cursor: pointer;
                transition: all 0.3s ease;
            }
            
            .bapopup-close:hover {
                background: #e0e0e0;
                transform: rotate(90deg);
            }

Overlay

.bapopup-overlay {
                background: rgba(0, 0, 0, 0.7);
                backdrop-filter: blur(3px);
            }

JavaScript Customization

// Custom close action
                jQuery('.bapopup-close').on('click', function() {
                    // Your custom code
                    console.log('Popup closed');
                });
                
                // Custom open animation
                jQuery('.bapopup-container').addClass('custom-animation');
