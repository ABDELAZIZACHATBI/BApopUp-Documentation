JavaScript API

Available Methods

showPopup()

Manually show a popup

// Show popup by ID
            bapopup.showPopup(123);

hidePopup()

Manually hide a popup

// Hide current popup
            bapopup.hidePopup();

Events

// Listen for popup events
            document.addEventListener('bapopup:shown', function(e) {
                console.log('Popup shown:', e.detail.popupId);
            });
            
            document.addEventListener('bapopup:hidden', function(e) {
                console.log('Popup hidden:', e.detail.popupId);
            });
