JavaScript API

Available Methods

BApopUp.show(popupId)

Manually show a popup

// Show popup with ID 123
            BApopUp.show(123);

BApopUp.hide()

Hide current popup

// Hide current popup
            BApopUp.hide();

BApopUp.clearCookies()

Clear all popup cookies

// Clear all popup cookies
            BApopUp.clearCookies();

BApopUp.on(event, callback)

Listen for popup events

// Listen for shown event
                            BApopUp.on('shown', function(e) {
                                console.log('Popup shown:', e.detail.popupId);
                            });
                            
                            // Listen for hidden event
                            BApopUp.on('hidden', function(e) {
                                console.log('Popup hidden:', e.detail.popupId);
                            });
